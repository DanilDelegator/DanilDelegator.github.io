GitHub Course CodeQL U-Boot

Those are my solutions to GitHub Course CodeQL U-Boot, posting them here because I felt like they are going to be good as a cheat sheet.

Solutions

Get all function definitions with the name strlen

import cpp

from Function f
where f.getName() = "strlen"
select f, "a fucntion named strlen"

Get all the macro definitions with a regex ntohs|ntohl|ntohll

import cpp

from Macro m
where m.getName().regexpMatch("ntohs|ntohl|ntohll")
select m, "Macros."

Get all function calls of the functions that are named memcpy

import cpp

from FunctionCall fc
// where f.getName() = "memcpy" and fc.getTarget() = f
where fc.getTarget().getName() = "memcpy"
select fc, "Call to memcpy function."

Get expressions of all macro invocations of the ntoh.*

import cpp

from MacroInvocation mi
where mi.getMacroName().regexpMatch("ntoh.*")
select mi.getExpr(), "ntoh* MacroInvocations Top Expression."

Get all places where a variable is used as the len argument in a memcpy function call after passing throw ntoh.* macro.

import cpp
import semmle.code.cpp.dataflow.TaintTracking
import DataFlow::PathGraph

class NetworkByteSwap extends Expr {
    NetworkByteSwap () {
        exists(MacroInvocation mi | mi.getMacroName().regexpMatch("ntoh.*") and mi.getExpr() = this)
    }
}

class Config extends TaintTracking::Configuration {
  Config() { this = "NetworkToMemFuncLength" }

  override predicate isSource(DataFlow::Node source) {
    source.asExpr() instanceof NetworkByteSwap
  }
  override predicate isSink(DataFlow::Node sink) {
    exists(FunctionCall fc | fc.getTarget().getName().regexpMatch("memcpy")
                         and fc.getArgument(2) = sink.asExpr())
  }
}

from Config cfg, DataFlow::PathNode source, DataFlow::PathNode sink
where cfg.hasFlowPath(source, sink)
select sink, source, sink, "Network byte swap flows to memcpy"
